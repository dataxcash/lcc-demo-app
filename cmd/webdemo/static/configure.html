// ProcessDataAnalytics performs advanced ML-based data analysis
func ProcessDataAnalytics(ctx context.Context, dataset *Dataset) error {
    // Check license before expensive operation
    allowed, remaining, reason, err := lccClient.Consume(
        "advanced_analytics", // Feature ID
        1,                        // Consume 1 credit
        nil,                     // Optional metadata
    )
    
    if err != nil {
        return fmt.Errorf("license check failed: %w", err)
    }
    
    if !allowed {
        // License doesn't permit this operation
        log.Warn("Advanced analytics denied", "reason", reason)
        return ErrFeatureNotAvailable
    }
    
    // License check passed, perform analysis
    result := analytics.RunMLModel(dataset)
    log.Info("Analytics completed", "remaining", remaining)
    
    return nil
}

// ExportReport generates PDF with rate limiting
func ExportReport(reportID string) error {
    // Rate limit check: prevent API abuse
    allowed, _, reason, err := lccClient.Consume(
        "pdf_export",
        1,
        map[string]any{"report_id": reportID},
    )
    
    if !allowed {
        if reason == "quota_exceeded" {
            return ErrDailyQuotaExceeded
        }
        return fmt.Errorf("export denied: %s", reason)
    }
    
    // Generate PDF (expensive operation)
    pdf := generator.CreatePDF(reportID)
    return pdf.Save()
}

// CreateProject enforces capacity limits
func CreateProject(name string) (*Project, error) {
    currentCount := db.CountProjects()
    
    // Check if we can create more projects
    allowed, maxCapacity, reason, err := lccClient.CheckCapacity(
        "capacity.project.count",
        currentCount + 1, // Desired new count
    )
    
    if !allowed {
        return nil, fmt.Errorf(
            "project limit reached: %d/%d (%s)",
            currentCount, maxCapacity, reason,
        )
    }
    
    // Create project
    project := &Project{Name: name}
    db.Save(project)
    
    return project, nil
}
